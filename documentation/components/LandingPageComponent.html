import { Component, OnInit, AfterViewInit } from '@angular/core';
import {Router} from '@angular/router';
import { CrudService } from '../../crud.service';
import * as _ from 'lodash';
declare var $: any;
@Component({
  selector: 'app-landing-page',
  templateUrl: './landing-page.component.html',
  styleUrls: ['./landing-page.component.css']
})
export class LandingPageComponent implements OnInit, AfterViewInit {

  private spinners:any = {first:false,second:false,third:false}
  private recommendedVideos: Array<any> = [];
  private topViewedVideos: Array<any> = [];
  private trendingVideos: Array<any> = [];
  private todaysTopViewed: Array<any> = [];
  private popularCreators: Array<any> = [];
  private popularCategories: Array<any> = [];
  private highlyInteractiveVideos: Array<any> = [];

  private todaysTop_1: any = {};
  private todaysTop_234: Array<any> = [];
  private topViewed_1: any = {};
  private topViewed_234: Array<any> = [];
  private topViewedByCategory: Array<any> = [];

  SlideOptions = { items: 5, dots: false, nav: true };  
  CarouselOptions = { items: 6, dots: true, nav: true };
  
  // Slider with 3 slides
  SlideOptions3 = { items: 2, dots: false, nav: true };  
  CarouselOptions3 = { items: 6, dots: true, nav: true };

  SlideOptions1 = { items: 1, dots: false, nav: true };

  constructor(private apiService: CrudService, 
    private router: Router) {

    }

  ngOnInit() {
    this.spinners.first = true;
    this.spinners.second = true;
    this.spinners.third = true;
    // this.spinners.fourth = true;
    this.getData();
  }

  getData() {
    this.apiService.getLandingPageData().subscribe(
      success => {

        // Top Viewed
        let arrTopViewed: Array<any> = success.data.topViewedVideos;
        arrTopViewed = _.sortBy(arrTopViewed, ['viewCount']);
        arrTopViewed = arrTopViewed.reverse();
        this.topViewedVideos = _.take(arrTopViewed, 6);
        this.topViewed_1 = this.topViewedVideos[0];
        this.spinners.first = false;
        
        //Recommended
        let arrRecommended: Array<any> = success.data.recommendedVideos; 
        arrRecommended = _.sortBy(arrRecommended, ['updatedAt']);
        arrRecommended = arrRecommended.reverse();
        this.recommendedVideos = _.take(arrRecommended, 6);
        this.spinners.second = false;
        
        // Trending
        let arrTrending: Array<any> = success.data.topViewedVideos;
        arrTrending = _.sortBy(arrTrending, ['viewCount']);
        arrTrending = arrTrending.reverse();
        this.trendingVideos = _.take(arrTrending, 5);

        // Popular categories
        let arrPopCategories: Array<any> = success.data.popularCategories;
        arrPopCategories = _.sortBy(arrPopCategories, ['totalVideos']);
        arrPopCategories = arrPopCategories.reverse();
        this.popularCategories = arrPopCategories;
        let maxViewedIn: Array<any> = arrPopCategories.map((category)=>category.maxViewedVideo.data);
        for (let i of maxViewedIn) {
          for (let j of i) {
            this.topViewedByCategory.push(i[0]);
            this.topViewedByCategory.push(i[1]);
            break;
          }
        }
        
        // Popular Creators
        let arrPopCreators: Array<any> = success.data.popularCreators;
        arrPopCreators = _.sortBy(arrPopCreators, ['updatedAt']);
        arrPopCreators = arrPopCreators.reverse();
        this.popularCreators = _.take(arrPopCreators, 6);
        
        // Highly Interactive
        let arrHighlyInt: Array<any> = success.data.highlyInteractiveVideos;
        arrHighlyInt = _.sortBy(arrHighlyInt, ['totalchild']);
        arrHighlyInt = arrHighlyInt.reverse();
        this.highlyInteractiveVideos = _.take(arrHighlyInt, 6);
        
        // Today's top viewed
        let arrTodaysTop: Array<any> = success.data.todaysTopViewed;
        arrTodaysTop = _.sortBy(arrTodaysTop, ['viewCount']);
        arrTodaysTop = arrTodaysTop.reverse();
        this.todaysTopViewed = _.take(arrTodaysTop, 4);
        this.todaysTop_1 = this.todaysTopViewed[0];
        this.todaysTop_234 = _.takeRight(this.todaysTopViewed, 3);
        this.spinners.third = false;
      }
    );
  }

  // Show video
  showVideo(videoId: any) {
    this.router.navigate(['videos'], { queryParams: { video: videoId } });
  }

  // Show user profile
  showUser(userId: any) {
    this.router.navigate(['public-profile'], { queryParams: { user: userId } });
  }

  ngAfterViewInit() {
    $('.interact-slider-start').owlCarousel({
      loop:false,
      margin:15,
      nav:true,
      dots:false,
      responsive:{
          0:{
              items:1
          },
          768:{
            items:2,
            stagePadding: 0
          },
          992:{
            items:1,
            stagePadding: 110
          },
          1500:{
            items:2,
            stagePadding: 110
          },
      }
    });

    $('.interact-slider-5').owlCarousel({
      loop:false,
      margin:12,
      nav:true,
      dots:false,
      responsive:{
          0:{
              items:1
          },
          500:{
            items:2
          },

              items:3
          },

              items:5
          }
      }
    });

      loop:false,
      margin:15,
      nav:true,
      dots:false,
      responsive:{
        0:{
            items:1
        },
        800:{
          items:1,
          stagePadding: 100
        },
        1100:{
          items:2,
          stagePadding: 100
        },
        // 800:{
        //     items:3
        // }
      }
    });
  }

}
