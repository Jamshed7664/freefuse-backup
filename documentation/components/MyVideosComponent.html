import { Component, OnInit } from '@angular/core';
import { CrudService } from '../../crud.service';
import { Router } from '@angular/router';
import { ToastrService } from 'ngx-toastr';
import * as _ from 'lodash';
declare var $;
@Component({
  selector: 'app-my-videos',
  templateUrl: './my-videos.component.html',
  styleUrls: ['./my-videos.component.css']
})
export class MyVideosComponent implements OnInit {

  private mainVideoList: Array<any> = [];
  private publishedVideos: Array<any> = [];
  private drafts: Array<any> = [];
  private intialDraft: Array<any> = [];
  private draftChunks: Array<any> = [];
  private userDetails: Array<any> = [];
  private contributors: Array<any> = [];
  private spinner: any =  false
  private val:any = 'recent';
  constructor(
    private crudService: CrudService,
    private toaster: ToastrService, 
    private router: Router) {

    }

  ngOnInit() {
    this.spinner = true;
    this.getMainVideos();
  }

  // Get all main videos
  getMainVideos() {
    this.intialDraft = [];
    this.mainVideoList = [];
    this.publishedVideos = [];
    this.drafts = [];
    this.crudService.getInteractiveVideo().subscribe(success => {
      this.mainVideoList = success.data;
      this.publishedVideos = this.mainVideoList.filter((video) => video.isPublished == true);
      this.drafts = this.mainVideoList.filter((video) => video.isPublished == false);
      for(let i=0;i<=2;i++)
      {
        if(this.drafts[i])
        this.intialDraft.push(this.drafts[i])
      
      }
      this.spinner = false;
      //  this.draftChunks =  this.splitDrafts();
    }, error => {});
  }

  // Get contributors

  viewAll()
  {
    this.intialDraft = this.drafts;
  }

  // splitDrafts(): Array<any>{
  //   let draft_chunk = _.chunk(this.drafts, 3);
  //   return draft_chunk;
  // }

  showVideo(id: any) {
   
    this.router.navigate(['videos'], { queryParams: { video: id } });
  
  }

  edit(id: any) {
    this.router.navigate(['create-interactive-video'], { queryParams: { video: id } });
  }

  showModal(id: any) {
    console.log('video id: ', id);
    console.log('Published Videos: ', this.publishedVideos);
    for (let i of this.mainVideoList) {
      if (id == i._id) {
        console.log('Contributors: ', i.contributors);
        this.contributors = i.contributors;
      }
    }
    $('#exampleModal').modal('show');
  }

  publish(id:any)
  {
    this.spinner = true;
    let finalObj = {
      "type":"publish",
      "videoID":id
      }
    this.crudService.publishunpublish(finalObj).subscribe(
      success => {
        this.getMainVideos();
      //this.createInteractiveVideoObj.spinner = false;
        this.toaster.success('Published successfully!');
      })
  }
  delete(id:any)
  {
    let confirm1 = confirm('Are you sure you want to delete?');
    if(confirm1){
    this.spinner = true;

    this.crudService.deleteBookMarkVideo(id).subscribe(
      success => {
        this.getMainVideos();
      //this.createInteractiveVideoObj.spinner = false;
        this.toaster.success('Deleted successfully!');
      })
    }
  }
  changeFilter(val:any){
    if(val =='recent')
    {
      this.publishedVideos.sort(function (a, b) {
        if (a.publishTime > b.publishTime) {
          return -1;
        }
        return 1
      })
    }
    else if(val =='viewed')
    {
      this.publishedVideos.sort(function (a, b) {
        if (a.viewCount > b.viewCount) {
          return -1;
        }
        return 1
      })
    }
    else if(val =="interactive")
    {
      this.publishedVideos.sort(function (a, b) {
        if (a.totalchild > b.totalchild) {
          return -1;
        }
        return 1
      })
      
    }
    else
    {
      this.publishedVideos.sort(function (a, b) {
        if (a.comments.length > b.comments.length) {
          return -1;
        }
        return 1
      })
    }
    
  }
  
  
}
